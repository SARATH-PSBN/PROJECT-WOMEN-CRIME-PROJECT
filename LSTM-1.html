# Sliding Window Function
def create_sequences(series,
                     sequence_length):

    X=[]

    y=[]

    for i in range(
        len(series)-sequence_length
    ):

        X.append(
            series[i:i+sequence_length]
        )

        y.append(
            series[i+sequence_length]
        )

    return np.array(X),np.array(y)

# Encapsulate in a Reusable Class

class CrimePreprocessor:
    def __init__(self, sequence_length=3):
        self.sequence_length = sequence_length
        self.scalers = {}

    def normalize_group(self, group):
        scaler = MinMaxScaler()
        group = group.copy()

        group["Crime Count"] = scaler.fit_transform(
            group[["Crime Count"]]
        )

        key = (
            group["STATE/UT"].iloc[0],
            group["Crime Head"].iloc[0]
        )

        self.scalers[key] = scaler
        return group

    def create_sequences(self, values):
        X, y = [], []

        for i in range(len(values)-self.sequence_length):
            X.append(values[i:i+self.sequence_length])
            y.append(values[i+self.sequence_length])

        return np.array(X), np.array(y)
